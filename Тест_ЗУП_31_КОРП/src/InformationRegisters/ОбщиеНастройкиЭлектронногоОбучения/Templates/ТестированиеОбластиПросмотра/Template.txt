    function test(data) {
        return data;
    }

    function listener(event) {

        try {
            // Возвращаем данные родителю
            window.parent.postMessage(event.data, "*");
        } catch (err) {
            alert(err)
        }

    }

    if (window.addEventListener) {
        window.addEventListener("message", listener);
    } else {
        // IE8
        window.attachEvent("onmessage", listener);
    }

</script>
